/* --------- Group by --------- */

1) Conta gli ospiti raggruppandoli per anno di nascita
SELECT
    COUNT(id) AS 'somma',
    YEAR(date_of_birth) AS 'anno'
FROM
    `ospiti`
GROUP BY
    YEAR(date_of_birth)


2) Somma i prezzi dei pagamenti raggruppandoli per status
SELECT
    SUM(price) AS 'somma',
    `status` AS 'status'
FROM
    pagamenti
GROUP BY
    `status`


3) Conta quante volte è stata prenotata ogni stanza
SELECT
    COUNT(id) AS 'n pren',
    `stanza_id` AS 'n stanza'
FROM
    prenotazioni
GROUP BY
    `stanza_id`


4) Fai una analisi per vedere se ci sono ore in cui le prenotazioni sono più frequenti
SELECT COUNT(id), HOUR(created_at)
FROM prenotazioni
GROUP BY HOUR(created_at)



5) Quante prenotazioni ha fatto l’ospite che ha fatto più prenotazioni?
SELECT
    COUNT(id) AS 'prenotazioni',
    ospite_id
FROM
    prenotazioni_has_ospiti
GROUP BY
    ospite_id
ORDER BY
    prenotazioni DESC

6) Come si chiamano gli ospiti che hanno fatto più di due prenotazioni?
SELECT
    COUNT(
        prenotazioni_has_ospiti.ospite_id
    ) AS 'tot_pren',
    ospiti.name 'Nome',
    ospiti.lastname 'Cognome'
FROM
    `ospiti`
INNER JOIN `prenotazioni_has_ospiti` ON ospiti.id = prenotazioni_has_ospiti.ospite_id
GROUP BY
    ospiti.name
HAVING
    COUNT(
        prenotazioni_has_ospiti.ospite_id
    ) > 2;
